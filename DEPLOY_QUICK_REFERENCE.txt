📋 部署前检查清单
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
☐ GitHub 账号已创建
☐ Git 已安装
☐ Node.js 18+ 已安装
☐ 项目在本地可以运行（npm run dev）

🔧 第一步：配置项目
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
编辑 vite.config.ts，在 defineConfig 中添加：

  base: '/claude-code-tutorial-zh/',

⚠️  重要：前后都要有斜杠！

🧪 第二步：本地测试
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
npm install
npm run build
npm run preview

访问 http://localhost:4173 验证页面正常显示

🌐 第三步：创建 GitHub 仓库
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. 访问 https://github.com/new
2. 仓库名：claude-code-tutorial-zh
3. 选择 Public
4. 不初始化任何文件
5. 点击 Create repository

📤 第四步：推送代码到 GitHub
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
git init
git add .
git commit -m "Initial commit"
git remote add origin https://github.com/YOUR_USERNAME/claude-code-tutorial-zh.git
git branch -M main
git push -u origin main

⚠️  替换 YOUR_USERNAME 为你的 GitHub 用户名

🚀 第五步：启用 GitHub Pages
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. 进入仓库 → Settings → Pages
2. Source: Deploy from a branch
3. Branch: main / (root)
4. 点击 Save

⏳ 第六步：等待部署
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
等待 2-5 分钟

✅ 第七步：验证部署
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
访问：https://YOUR_USERNAME.github.io/claude-code-tutorial-zh

检查：
  ☐ 页面正常加载（不是 404）
  ☐ 样式正确显示
  ☐ 导航链接可以点击
  ☐ 页面刷新后不会 404

🚀 使用部署脚本（可选）
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
macOS/Linux:
  chmod +x deploy.sh
  ./deploy.sh

Windows:
  deploy.bat

脚本会自动完成所有部署步骤

🔄 更新已部署的内容
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

git push origin main

GitHub Pages 会自动重新部署（1-2 分钟）

❓ 常见问题
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
Q: 页面显示 404？
A: 检查 vite.config.ts 的 base 路径是否正确

Q: 样式没有加载？
A: 清除浏览器缓存，检查 base 路径配置

Q: 路由不工作？
A: 已配置 404.html，清除缓存后重试

Q: 如何查看部署日志？
A: GitHub 仓库 → Actions 标签
